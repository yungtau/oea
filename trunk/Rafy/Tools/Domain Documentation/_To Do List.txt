	{
		public static Rafy.Data.IDbAccesser CreateDBA()

			return new Rafy.Data.DbAccesser(DbSetting.FindOrCreate("ConfigBase"));
		}

		public static void Demo()
		{
			using (var dba = CreateDBA())
			{
				var value = dba.QueryValue("select isnull(max(RootSort)+10,10) from SysSectorMg");
			}
		}
	}
